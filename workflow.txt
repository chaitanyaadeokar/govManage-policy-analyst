flowchart TD

%% =======================
%% PERCEPTION LAYER
%% =======================
A[Trigger Event<br/>Financial Txn / IT Incident / Policy Upload / Cron Job]

B[Preprocessor<br/>Clean + Validate + Normalize JSON]

A --> B

%% =======================
%% ORCHESTRATION
%% =======================
C[Orchestrator Agent<br/>LangGraph Router]

B --> C

%% =======================
%% PARALLEL AGENT EXECUTION
%% =======================
C --> D1[Policy Analyst Agent]
C --> D2[Compliance Agent]
C --> D3[Risk Assessment Agent]
C --> D4[Audit Agent]
C --> D5[Reporting Agent]

%% =======================
%% DATABASE INTERACTIONS
%% =======================
D1 --> DB1[(Policy_Embeddings / ChromaDB)]
D2 --> DB2[(Employees DB)]
D2 --> DB3[(Governance_Actions DB)]
D3 --> DB4[(Risk Parameters DB)]
D4 --> DB5[(Audit Logs DB)]
D5 --> DB6[(Reports DB)]

%% =======================
%% POLICY ANALYSIS LOGIC
%% =======================
D1 --> P1{Policy Violation?}

P1 -->|Yes| P2[Extract Rule + Conflict Score]
P1 -->|No| P3[Mark Policy OK]

%% =======================
%% COMPLIANCE LOGIC
%% =======================
D2 --> C1{Role Valid?}

C1 -->|No| C2[Clearance Mismatch]
C1 -->|Yes| C3[Compliance OK]

%% =======================
%% RISK ENGINE (TVI)
%% =======================
D3 --> R1[Compute TVI Score<br/>Threat x Vulnerability x Impact]

R1 --> R2{Risk Level}

R2 -->|Low| R3[Score < 20]
R2 -->|Medium| R4[20-70]
R2 -->|High| R5[>70]

%% =======================
%% MULTI-AGENT DECISION ENGINE
%% =======================
P2 --> J1
C2 --> J1
R5 --> J1

P3 --> J2
C3 --> J2
R3 --> J2

J1[Conflict / High Risk Detected]
J2[Safe Path]

%% =======================
%% HUMAN-IN-LOOP (AMBIGUITY)
%% =======================
J1 --> H1{Ambiguity? Policy vs Risk Conflict}

H1 -->|Yes| H2[Escalate to Human<br/>CISO Approval]
H1 -->|No| H3[Auto Block / Freeze]

%% =======================
%% ACTION ENGINE
%% =======================
J2 --> AE1[Approve Transaction]
H2 --> AE2[Wait for Human Decision]
H3 --> AE3[Revoke Access / Freeze Txn]

%% =======================
%% ACTION OUTPUTS
%% =======================
AE1 --> O1[Update Governance_Actions = Approved]
AE3 --> O2[Trigger API<br/>Revoke Access / Alerts]
AE2 --> O3[Manual Override Logged]

%% =======================
%% AUDIT (IMMUTABLE)
%% =======================
O1 --> ALOG
O2 --> ALOG
O3 --> ALOG

ALOG[Audit Agent<br/>Immutable Log + Reasoning Trace]

%% =======================
%% REPORTING
%% =======================
ALOG --> REP[Reporting Agent<br/>Executive Summary]

REP --> DASH[Dashboard<br/>KPIs + Risk Index]

%% =======================
%% FEEDBACK LOOP (SELF-HEALING)
%% =======================
ALOG --> FB[Feedback & Improvement Agent]

FB --> F1{Repeated Violations?}

F1 -->|Yes| F2[Optimize Policy<br/>Update Thresholds]
F1 -->|No| F3[No Change]

F2 --> DB1
F2 --> DB4

%% =======================
%% EXTERNAL THREAT INGESTION
%% =======================
EXT[External Feed<br/>News / Cyber Alerts]

EXT --> C
